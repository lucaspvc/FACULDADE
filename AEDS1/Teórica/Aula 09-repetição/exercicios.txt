Exercício 1
calculo de uma fatorial

algoritmo
    declare N, FAT numerico
    leia N
    FAT <- 1
    repita
        se N < 1
            então interrompa
        fim-se
        FAT <- FAT * N 
        N <- N-1
    fim-repita
    escreva "Fatorial é: FAT"
fim-algoritmo

Exercício 2 

algoritmo
    declare PESO, MAIOR, MENOR, ID, IDMAIOR, IDMENOR, I numerico
    leia PESO, ID
    MAIOR <- PESO
    MENOR <- PESO
    I = 1
    repita
        se I > 90
            então interrompa
        fim-se
        senão
            leia PESO, ID
            I <- I + 1
            se MAIOR < PESO
                então MAIOR <- PESO 
                    IDMAIOR <- ID
            fim-se   
        senão 
            se MENOR > PESO
                então MENOR <- PESO
                    IDMENOR <- ID
            fim-se
        fim-senão
        fim-senão
    escreva IDMAIOR, MAIOR
    escreva IDMENOR, MENOR
fim-algoritmo
